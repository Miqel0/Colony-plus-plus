192 ImGui::Separator();
193
194 if (!dane.empty()) {
195 if (ImGui::BeginTable("StatsTooltip", 2, ImGuiTableFlags_BordersInnerH)) {

198 ImGui::TableNextColumn();

211

223
224 sf::Sprite ikonka(*globalnaTeksturaIkonek);

227 float offsetX = (ImGui::GetColumnWidth() - 16.0f) * 0.5f;

237 ImGui::Text("%s", para.first.c_str());

241 ImGui::TableNextColumn();
